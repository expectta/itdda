'use strict'; const { Model } = require('sequelize'); module.exports = (sequelize, DataTypes) => { class users_institutions extends Model { static associate(models) { models.users_institutions.belongsTo(models.user); models.users_institutions.belongsTo(models.institution); } } users_institutions.init( { id: { allowNull: false, autoIncrement: true, primaryKey: true, type: DataTypes.INTEGER, }, }, { sequelize, modelName: 'users_institutions', }, ); return users_institutions; };
